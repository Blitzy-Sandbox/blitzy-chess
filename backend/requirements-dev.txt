# Development & test dependencies for the chess_ai backend.
#
# Installed by `make init` (the init-backend target) ALONGSIDE the runtime
# manifest, both passed explicitly:
#     pip install -r requirements.txt -r requirements-dev.txt
#
# This file is intentionally STANDALONE: it lists dev/test-only packages and
# deliberately does NOT `-r requirements.txt`. The two manifests are installed
# together by the Makefile, so an include here would be redundant and would
# double-resolve the runtime set.
#
# Version policy (matches requirements.txt): `>=` floors come from the project
# prompt and are treated as canonical minimums; exact pins use `==`.

# --- Testing ----------------------------------------------------------------
# pytest         : test runner for backend/tests/  (make test-backend).
# pytest-cov     : coverage reporting (opt-in: pytest --cov=chess_ai).
# pytest-asyncio : runs the async WebSocket suites (test_game_ws.py,
#                  test_multiplayer_ws.py, test_self_play.py). pyproject.toml
#                  sets asyncio_mode = "auto" so no per-test marker is needed.
# httpx          : async / WebSocket-capable ASGI client for the endpoint tests.
pytest>=8.0
pytest-cov>=5.0
pytest-asyncio>=0.23
httpx>=0.27

# --- Lint & format ----------------------------------------------------------
# ruff : backs `make lint` (ruff check .) and `make format` (ruff format .).
ruff>=0.4

# --- Browser automation (self-play demonstration — Constraint 14) -----------
# playwright drives the headless browser for the self-play runner and
# backend/tests/test_self_play.py. The browser binary is NOT a Python package;
# `make init` installs it separately via `playwright install chromium`.
playwright==1.60.0
